  3 /**
  4  * The Manager acts as a container for all widgets. 
  5  * It stores Solr configuration and selection and delegates calls to the widgets.
  6  * All public calls should be performed on the manager object.
  7  *
  8  * @param properties A map of fields to set. Refer to the list of public fields.
  9  * @class AbstractManager
 10  */
 11 AjaxSolr.AbstractManager = AjaxSolr.Class.extend(
 12   /** @lends AjaxSolr.AbstractManager.prototype */
 13   {
 14   /** 
 15    * The absolute URL to the Solr instance.
 16    *
 17    * @field
 18    * @public
 19    * @type String
 20    * @default http://localhost:8983/solr/select/
 21    */
 22   solrUrl: 'http://localhost:8983/solr/select/',
 23 
 24   /**
 25    * If we want to proxy queries through a script, rather than send queries
 26    * to Solr directly, set the passthruUrl field to the fully-qualified URL.
 27    *
 28    * @field
 29    * @public
 30    * @type String
 31    */
 32   passthruUrl: null,
 33 
 34   /**
 35    * Filters to apply to all queries.
 36    *
 37    * @field
 38    * @public
 39    */
 40   filters: {
 41     q: [],
 42     fq: [],
 43     fl: []
 44   },
 45 
 46   /**
 47    * The field to highlight when rendering results.
 48    *
 49    * @field
 50    * @public
 51    * @type String
 52    * @default "body"
 53    */
 54   hlFl: 'body',
 55 
 56   /** 
 57    * A collection of all registered widgets. For internal use only.
 58    *
 59    * @field
 60    * @private 
 61    * @default {}
 62    */
 63   widgets: {},
 64 
 65   /** 
 66    * The Solr start offset parameter.
 67    *
 68    * @field
 69    * @private 
 70    * @type Number
 71    * @default 0
 72    */
 73   start: 0,
 74 
 75   /**
 76    * A copy of the URL hash, so we can detect any changes to it.
 77    *
 78    * @field
 79    * @private
 80    * @type String
 81    * @default ""
 82    */
 83   hash: '',
 84 
 85   /** 
 86    * Adds a widget to the manager.
 87    *
 88    * @param {AjaxSolr.AbstractWidget} widget
 89    */
 90   addWidget: function(widget) { 
 91     if (this.canAddWidget(widget)) {
 92       widget.manager = this;
 93       this.widgets[widget.id] = widget;
 94       widget.afterAdditionToManager();
 95     }
 96   },
 97 
 98   /**
 99    * An abstract hook for child implementations.
100    *
101    * @param {AjaxSolr.AbstractWidget} widget
102    * @returns {Boolean} Whether the DOM is ready for the widget.
103    */
104   canAddWidget: function(widget) {
105     return true;
106   },
107 
108   /**
109    * Initializes the manager.
110    *
111    * Loads the query from the hash, submits a request, and adds hash change
112    * listeners to submit requests if the hash changes, e.g. back button click.
113    */
114   init: function() {
115     this.loadQueryFromHash();
116     this.doInitialRequest();
117 
118     // Support the back button.
119     var me = this;
120     window.setInterval(function() {
121       if (window.location.hash.length) {
122         if (me.hash != window.location.hash) {
123           me.loadQueryFromHash();
124           me.doInitialRequest();
125         }
126       }
127       // Without this condition, the user is not able to back out of search.
128       else {
129         history.back();
130       }
131     }, 250);
132   },
133 
134   /** 
135    * Adds the given items to the given widget, and runs the request.
136    *
137    * @param {String} widgetId The id of the widget.
138    * @param {Array} items The items to select.
139    */
140   selectItems: function(widgetId, items) {
141     if (this.widgets[widgetId].selectItems(items)) {
142       this.doRequest(0);
143     }
144   },
145 
146   /** 
147    * Removes the given items from the given widget, and runs the request.
148    *
149    * @param {String} widgetId The id of the widget.
150    * @param {Array} items The items to deselect.
151    */  
152   deselectItems: function(widgetId, items) {
153     if (this.widgets[widgetId].deselectItems(items)) {
154       this.doRequest(0);
155     }
156   },
157 
158   /**
159    * Removes all items from the given widget, and runs the request.
160    *
161    * @param {String} widgetId The id of the widget.
162    */
163   deselectWidget: function(widgetId) {
164     this.widgets[widgetId].deselectAll();
165     this.doRequest(0);
166   },
167 
168   /**
169    * Removes all items from all widgets except the given widget, adds the given
170    * items to the given widget, and runs the request.
171    *
172    * @param {String} keepId The id of the widget.
173    * @param {Array} items The items to select.
174    */
175   selectOnlyItems: function(keepId, items) {
176     for (var widgetId in this.widgets) {
177       if (widgetId == keepId) {
178         this.widgets[keepId].selectItems(items);
179       }
180       else {
181         this.widgets[widgetId].deselectAll();
182       }
183     }
184     this.doRequest(0);
185   },
186 
187   /**
188    * Removes all items from all widgets except the given widget, and runs the
189    * request.
190    *
191    * @param {String} keepId The id of the widget.
192    */
193   selectOnlyWidget: function(keepId) {
194     for (var widgetId in this.widgets) {
195       if (widgetId != keepId) {
196         this.widgets[widgetId].deselectAll();
197       }
198     }
199     this.doRequest(0);
200   },
201 
202   /**
203    * Removes all items from all widgets, and runs the request.
204    */
205   deselectAll: function() {
206     for (var widgetId in this.widgets) {
207       this.widgets[widgetId].deselectAll();
208     }
209     this.doRequest(0);
210   },
211 
212   /**
213    * Loads the query from the URL hash.
214    */
215   loadQueryFromHash: function() {
216     // if the hash is empty, the page must be loading for the first time,
217     // so don't clobber items selected during afterAdditionToManager().
218     if (window.location.hash.length) {
219       for (var widgetId in this.widgets) {
220         this.widgets[widgetId].deselectAll();
221       }
222     }
223 
224     var hash = window.location.hash.substring(1);
225     var vars = hash.split('&');
226 
227     for (var i = 0; i < vars.length; i++) {
228       if (vars[i].substring(0, 3) == 'fq=') {
229         var item = new AjaxSolr.FilterQueryItem();
230         item.parseHash(vars[i].substring(3));
231 
232         if (this.widgets[item.widgetId]) {
233           this.widgets[item.widgetId].selectItems([ item.value ]);
234         }
235       }
236       else if (vars[i].substring(0, 2) == 'q=') {
237         var item = new AjaxSolr.QueryItem();
238         item.parseHash(vars[i].substring(2));
239 
240         if (this.widgets.text) {
241           this.widgets.text.selectItems([ item.value ]);
242         }
243       }
244       else if (vars[i].substring(0, 6) == 'start=') {
245         this.start = parseInt(vars[i].substring(6));
246       }
247     }
248   },
249 
250   /**
251    * Stores the query in the URL hash.
252    *
253    * @param queryObj The query object built by buildQuery.
254    */
255   saveQueryToHash: function(queryObj) {
256     var hash = '#';
257     for (var i in queryObj.fq) {
258       hash += 'fq=' + queryObj.fq[i].toHash() + '&';
259     }
260     for (var i in queryObj.q) {
261       hash += 'q=' + queryObj.q[i].toHash() + '&';
262     }
263     hash += 'start=' + queryObj.start;
264 
265     window.location.hash = hash;
266 
267     // Don't assign this.hash to hash as window.location.hash undergoes some
268     // internal processing after assignment. Assign it to window.location.hash
269     // after setting window.location.hash to ensure the two are equal.
270     this.hash = window.location.hash;
271   },
272 
273   /**
274    * Returns an object decorated with Solr parameters, e.g. q, fl, fq, start,
275    * rows, fields, dates, sort, etc. Used in alterQuery(), displayQuery(),
276    * executeRequest(), and saveQueryToHash().
277    *
278    * @param {Number} start The Solr start offset parameter.
279    * @returns The query object.
280    */
281   buildQuery: function(start) {
282     var queryObj = {
283       fields: [],
284       dates: []
285     };
286 
287     queryObj.q = this.filters.q.slice();
288     queryObj.fl = this.filters.fl.slice();
289     queryObj.fq = this.filters.fq.slice();
290     queryObj.start = start;
291     queryObj.rows = 0;
292 
293     for (var widgetId in this.widgets) {
294       this.widgets[widgetId].alterQuery(queryObj);
295     }
296 
297     return queryObj;
298   },
299 
300   /**
301    * Transforms a query object into a string for execution.
302    *
303    * @param queryObj The query object built by buildQuery.
304    * @param {Boolean} skip Whether to skip URL encoding.
305    * @returns {String} The query object as a string.
306    */
307   buildQueryString: function(queryObj, skip) {
308     // Basic facet info. Return the top 40 items for each facet and ignore anything with 0 results
309     var query = 'facet=true&facet.limit=40&facet.sort=true&facet.mincount=1&hl=true';
310 
311     // Fields is the list of facets that will have counts and such returned
312     for (var i in queryObj.fields) {
313       query += '&facet.field=' + queryObj.fields[i].urlencode(skip);
314     }
315 
316     for (var i in queryObj.dates) {
317       var field = queryObj.dates[i].field;
318       query += '&facet.date=' + field.urlencode(skip);
319       query += '&f.' + field + '.facet.date.start=' + queryObj.dates[i].start.urlencode(skip);
320       query += '&f.' + field + '.facet.date.end=' + queryObj.dates[i].end.urlencode(skip);
321       query += '&f.' + field + '.facet.date.gap=' + queryObj.dates[i].gap.urlencode(skip);
322     }
323 
324     // Solr uses fq for facet based searching
325     for (var i in queryObj.fq) {
326       query += '&fq=' + queryObj.fq[i].toSolr(skip);
327     }
328 
329     // Solr uses q for free text searching
330     var q = '';
331     for (var i in queryObj.q) {
332       q += queryObj.q[i].toSolr(skip) + ' ';
333     }
334     query += '&q=' + q;
335 
336     queryObj.fl.push('id');
337 
338     query += '&fl=' + queryObj.fl.join(',');
339     query += '&rows=' + queryObj.rows;
340     query += '&start=' + queryObj.start;
341     if (queryObj.sort) {
342       query += '&sort=' + queryObj.sort;
343     }
344 
345     query += '&hl.fl=' + this.hlFl;
346 
347     return query;
348   },

350   /** 
351    * Creates a query out of the current selection, starts any widget loading
352    * animations, display the query, request the data from the Solr server, and
353    * saves the query to the URL hash.
354    *
355    * @param {Number} start The Solr start offset parameter.
356    */
357   doRequest: function(start) {
358     var queryObj = this.buildQuery(start);
359 
360     for (var widgetId in this.widgets) {
361       this.widgets[widgetId].startAnimation();
362     }
363 
364     for (var widgetId in this.widgets) {
365       this.widgets[widgetId].displayQuery(queryObj);
366     }
367 
368     this.executeRequest(queryObj);
369 
370     this.saveQueryToHash(queryObj);
371   },
372 
373   /**
374    * Calls doRequest() with the current start offset.
375    */
376   doInitialRequest: function() {
377     this.doRequest(this.start);
378   },
379 
380   /**
381    * An abstract hook for child implementations.
382    * Sends the request to Solr and handles the response.
383    * Should use jsonCallback() to handle the request.
384    *
385    * @param queryObj The query object built by buildQuery.
386    * @throws If not defined in child implementation.
387    */
388   executeRequest: function(queryObj) {
389     throw 'Abstract method executeRequest';
390   },
391 
392   /**
393    * Returns the callback to feed to, e.g. jQuery.getJSON or jQuery.post.
394    *
395    * @returns {Function}
396    */
397   jsonCallback: function() {
398     var me = this;
399     return function(data) {
400       me.handleResult(data);
401     }
402   },
403 
404   /**
405    * This method is executed after the Solr response data arrives. Passes the
406    * Solr response to the widgets, for each widget to handle separately, and
407    * ends any widget loading animations.
408    *
409    * @param data The Solr response inside a JavaScript object.
410    */
411   handleResult: function(data) {
412     // for debugging purposes
413     this.responseCache = data;
414 
415     for (var widgetId in this.widgets) {
416       this.widgets[widgetId].handleResult(data);
417     }
418 
419     for (var i in this.widgets) {
420       this.widgets[i].endAnimation();
421     }
422   }
