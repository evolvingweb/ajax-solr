  3 /**
  4  * The Manager acts as the controller in a Model-View-Controller framework. All
  5  * public calls should be performed on the manager object.
  6  *
  7  * @param properties A map of fields to set. Refer to the list of public fields.
  8  * @class AbstractManager
  9  */
 10 AjaxSolr.AbstractManager = AjaxSolr.Class.extend(
 11   /** @lends AjaxSolr.AbstractManager.prototype */
 12   {
 13   /**
 14    * The absolute URL to the Solr select servlet.
 15    *
 16    * @field
 17    * @public
 18    * @type String
 19    * @default "http://localhost:8983/solr/select"
 20    */
 21   solrUrl: 'http://localhost:8983/solr/select',
 22 
 23   /**
 24    * If we want to proxy queries through a script, rather than send queries
 25    * to Solr directly, set the proxyUrl field to the fully-qualified URL.
 26    *
 27    * @field
 28    * @public
 29    * @type String
 30    */
 31   proxyUrl: null,
 32 
 33   /**
 34    * The most recent response from Solr.
 35    *
 36    * @field
 37    * @private
 38    * @type Object
 39    * @default {}
 40    */
 41   response: {},
 42 
 43   /** 
 44    * A collection of all registered widgets. For internal use only.
 45    *
 46    * @field
 47    * @private
 48    * @type Object
 49    * @default {}
 50    */
 51   widgets: {},
 52 
 53   /**
 54    * The parameter store for the manager and its widgets. For internal use only.
 55    *
 56    * @field
 57    * @private
 58    * @type Object
 59    */
 60   store: null,
 61 
 62   /**
 63    * Whether <tt>init()</tt> has been called yet. For internal use only.
 64    *
 65    * @field
 66    * @private
 67    * @type Boolean
 68    * @default false
 69    */
 70   initialized: false,
 71 
 72   /**
 73    * An abstract hook for child implementations.
 74    *
 75    * <p>This method should be called after the store and the widgets have been
 76    * added. It should initialize the widgets and the store, and do any other
 77    * one-time initializations, e.g., perform the first request to Solr.</p>
 78    *
 79    * <p>If no store has been set, it sets the store to the basic <tt>
 80    * AjaxSolr.ParameterStore</tt>.</p>
 81    */
 82   init: function () {
 83     this.initialized = true;
 84     if (this.store === null) {
 85       this.setStore(new AjaxSolr.ParameterStore());
 86     }
 87     this.store.load(false);
 88     for (var widgetId in this.widgets) {
 89       this.widgets[widgetId].init();
 90     }
 91     this.store.init();
 92   },
 93 
 94   /**
 95    * Set the manager's parameter store.
 96    *
 97    * @param {AjaxSolr.ParameterStore} store
 98    */
 99   setStore: function (store) { 
100     store.manager = this;
101     this.store = store;
102   },
103 
104   /** 
105    * Adds a widget to the manager.
106    *
107    * @param {AjaxSolr.AbstractWidget} widget
108    */
109   addWidget: function (widget) { 
110     widget.manager = this;
111     this.widgets[widget.id] = widget;
112   },
113 
114   /** 
115    * Stores the Solr parameters to be sent to Solr and sends a request to Solr.
116    *
117    * @param {Boolean} [start] The Solr start offset parameter.
118    */
119   doRequest: function (start) {
120     if (this.initialized === false) {
121       this.init();
122     }
123     // Allow non-pagination widgets to reset the offset parameter.
124     if (start !== undefined) {
125       this.store.get('start').val(start);
126     }
127 
128     this.store.save();
129 
130     for (var widgetId in this.widgets) {
131       this.widgets[widgetId].beforeRequest();
132     }
133 
134     this.executeRequest();
135   },
136 
137   /**
138    * An abstract hook for child implementations.
139    *
140    * <p>Sends the request to Solr, i.e. to <code>this.solrUrl</code> or <code>
141    * this.proxyUrl</code>, and receives Solr's response. It should send <code>
142    * this.store.string()</code> as the Solr query, and it should pass Solr's
143    * response to <code>handleResponse()</code> for handling.</p>
144    *
145    * <p>See <tt>managers/Manager.jquery.js</tt> for a jQuery implementation.</p>
146    *
147    * @throws If not defined in child implementation.
148    */
149   executeRequest: function () {
150     throw 'Abstract method executeRequest must be overridden in a subclass.';
151   },
152 
153   /**
154    * This method is executed after the Solr response data arrives. Allows each
155    * widget to handle Solr's response separately.
156    *
157    * @param {Object} data The Solr response.
158    */
159   handleResponse: function (data) {
160     this.response = data;
161 
162     for (var widgetId in this.widgets) {
163       this.widgets[widgetId].afterRequest();
164     }
165   }
166 });
